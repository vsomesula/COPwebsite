Find grocery sales total revenue

purchases %>%
    left_join(prices) %>%
    summarise(total_revenue = sum(price))
## Joining, by = "item"
## # A tibble: 1 x 1
##   total_revenue
##           <dbl>
## 1          5.75

Find grocery sales revenue per customer

purchases %>%
  left_join(prices) %>%
  group_by(customer_id) %>%
  summarise(total_revenue = sum(price))
## Joining, by = "item"
## # A tibble: 2 x 2
##   customer_id total_revenue
##         <dbl>         <dbl>
## 1           1          1.95
